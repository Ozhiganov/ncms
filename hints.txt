DB2 INTEGRATION

to enable offset and limit:

db2set DB2_COMPATIBILITY_VECTOR=4000
db2stop
db2start

db2 "CREATE DATABASE $DB AUTOMATIC STORAGE YES USING CODESET UTF-8 TERRITORY RU COLLATE USING SYSTEM PAGESIZE 16384"
db2 connect to nsusite
db2 "CREATE BUFFERPOOL BIGPAGEBP IMMEDIATE  SIZE 4096 AUTOMATIC PAGESIZE 32 K"
db2 "CREATE SYSTEM TEMPORARY TABLESPACE TEMPSPACE32 PAGESIZE 32 K MANAGED BY AUTOMATIC STORAGE BUFFERPOOL BIGPAGEBP"
db2 "CREATE SCHEMA nsusite AUTHORIZATION nsusite"
db2 "GRANT DBADM ON DATABASE TO USER nsusite"
db2 connect reset
